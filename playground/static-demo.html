📊 Static Router Configuration

// Assets folder
staticRouter.static('./assets', {
  cache: true,
  maxAge: 3600,        // 1 hour
  maxFileSize: 5MB,
  etag: true,
  gzip: true
});

// Multiple folders with different settings
staticRouter.static(['./images', './icons'], {
  cache: true,
  maxAge: 86400,       // 24 hours
  maxFileSize: 10MB,
  gzip: false          // Images are already compressed
});

// CSS with aggressive compression
staticRouter.static(['./css', './styles'], {
  cache: true,
  maxAge: 1800,        // 30 minutes
  gzip: true,
  brotli: true
});

// JavaScript with short cache
staticRouter.static('./js', {
  cache: true,
  maxAge: 900,         // 15 minutes
  gzip: true,
  brotli: true
});

// Fonts with long cache
staticRouter.static('./fonts', {
  cache: true,
  maxAge: 604800,      // 1 week
  gzip: false,
  brotli: false
});
